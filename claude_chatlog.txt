dax
Top_5_Traffic_Moving = 
VAR TrafficOfficers = 
    CALCULATETABLE(
        VALUES('ATS_Court_Data'[OFFICER_DISPLAY_NAME]),
        'ATS_Court_Data'[WG2] = "Traffic Bureau",
        'ATS_Court_Data'[TYPE] = "M",
        'ATS_Court_Data'[ASSIGNMENT_FOUND] = TRUE()
    )
VAR TopOfficersTable = 
    ADDCOLUMNS(
        TrafficOfficers,
        "MovingCount", 
        CALCULATE(
            COUNTROWS('ATS_Court_Data'),
            'ATS_Court_Data'[TYPE] = "M",
            'ATS_Court_Data'[WG2] = "Traffic Bureau"
        )
    )
VAR Top5Officers = 
    TOPN(5, TopOfficersTable, [MovingCount], DESC)
VAR Result = 
    CONCATENATEX(
        Top5Officers,
        [OFFICER_DISPLAY_NAME] & ": " & [MovingCount],
        UNICHAR(10)
    )
RETURN Result
